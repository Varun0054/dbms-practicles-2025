```sql
CREATE TABLE Stud_Marks (
  name VARCHAR2(100) PRIMARY KEY,
  total_marks NUMBER
);

CREATE TABLE Result (
  Roll NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  Name VARCHAR2(100),
  Class VARCHAR2(30)
);
```

```sql
CREATE OR REPLACE PROCEDURE proc_Grade(p_name IN VARCHAR2, p_total IN NUMBER) AS
  v_class VARCHAR2(30);
BEGIN
  IF p_total BETWEEN 990 AND 1500 THEN v_class := 'Distinction';
  ELSIF p_total BETWEEN 900 AND 989 THEN v_class := 'First Class';
  ELSIF p_total BETWEEN 825 AND 899 THEN v_class := 'Higher Second Class';
  ELSE v_class := 'Other'; END IF;
  INSERT INTO Result(Name, Class) VALUES (p_name, v_class);
END;
/

-- Caller block example
DECLARE
  v_rec Stud_Marks%ROWTYPE;
BEGIN
  FOR v_rec IN (SELECT * FROM Stud_Marks) LOOP
    proc_Grade(v_rec.name, v_rec.total_marks);
  END LOOP;
END;
/
```
