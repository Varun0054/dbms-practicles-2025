```sql
CREATE TABLE Stud (
  Roll NUMBER PRIMARY KEY,
  Name VARCHAR2(100),
  Attendance NUMBER(5,2),
  Status VARCHAR2(20)
);
```

```sql
DECLARE
  v_roll Stud.Roll%TYPE := :roll_in; -- bind variable/user input
  v_att  Stud.Attendance%TYPE;
BEGIN
  SELECT Attendance INTO v_att FROM Stud WHERE Roll = v_roll;
  IF v_att < 75 THEN
    UPDATE Stud SET Status = 'Detained' WHERE Roll = v_roll;
    DBMS_OUTPUT.PUT_LINE('Term not granted');
  ELSE
    UPDATE Stud SET Status = 'Not Detained' WHERE Roll = v_roll;
    DBMS_OUTPUT.PUT_LINE('Term granted');
  END IF;
EXCEPTION
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('Roll not found');
END;
/
```
